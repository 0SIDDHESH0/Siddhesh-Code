                    function prinum(){
                        var a = +document.getElementById("num").value;
                        var p = true;
                        for (let i = 2; i < a; i++) {
                            if (a % i == 0){
                                p = false;
                                break;
                            }
                        }
                        if (p == true) {
                            alert("It's a prime number");
                        } 
                        else {
                            alert("It's a not prime number");
                        }
            
                    }
